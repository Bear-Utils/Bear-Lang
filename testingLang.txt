# Functions
FUN oopify(prefix) -> prefix + "oop"

FUN join(elements, seperator)
    VAR result = ""
    VAR len = LEN(elements)

    FOR i = 0 TO len THEN
        VAR result = result + elements/i
        IF i != len - 1 THEN VAR result = result + seperator
    END

    RETURN result
END
